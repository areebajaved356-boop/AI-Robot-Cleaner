<script>
const rows = 6;
const cols = 6;
let room = [];
let robot = {x: 0, y: 0};
let interval;

// Initialize room
function initRoom() {
    room = [];
    for (let i = 0; i < rows; i++) {
        room[i] = [];
        for (let j = 0; j < cols; j++) {
            room[i][j] = Math.random() > 0.5 ? 1 : 0;
        }
    }
    robot = {x: 0, y: 0};
}

// Draw grid
function drawRoom() {
    const table = document.getElementById("room");
    table.innerHTML = "";

    for (let i = 0; i < rows; i++) {
        let rowHTML = "<tr>";
        for (let j = 0; j < cols; j++) {
            let cls = "clean";
            let text = "";

            if (i === robot.x && j === robot.y) {
                cls = "robot";
                text = "R";
            } else if (room[i][j] === 1) {
                cls = "dirty";
                text = "D";
            }

            rowHTML += `<td class="${cls}">${text}</td>`;
        }
        rowHTML += "</tr>";
        table.innerHTML += rowHTML;
    }
}

// Clean current cell
function clean() {
    if (room[robot.x][robot.y] === 1) {
        room[robot.x][robot.y] = 0;
    }
}

// BFS to find nearest dirt
function findNearestDirt() {
    let queue = [[robot.x, robot.y, []]];
    let visited = Array.from({length: rows}, () => Array(cols).fill(false));
    visited[robot.x][robot.y] = true;

    let directions = [[-1,0],[1,0],[0,-1],[0,1]];

    while (queue.length > 0) {
        let [x, y, path] = queue.shift();

        if (room[x][y] === 1) return path;

        for (let [dx, dy] of directions) {
            let nx = x + dx;
            let ny = y + dy;

            if (nx >= 0 && ny >= 0 && nx < rows && ny < cols && !visited[nx][ny]) {
                visited[nx][ny] = true;
                queue.push([nx, ny, [...path, [nx, ny]]]);
            }
        }
    }
    return null;
}

// Move robot
function moveTo(path) {
    if (path && path.length > 0) {
        let [nx, ny] = path.shift();
        robot.x = nx;
        robot.y = ny;
    }
}

// Start cleaning
function startCleaning() {
    clearInterval(interval);

    interval = setInterval(() => {
        clean();

        let path = findNearestDirt();
        if (!path) {
            clearInterval(interval);
            alert("All Cleaned ✅");
            return;
        }

        moveTo(path);
        drawRoom();
    }, 400);
}

// Reset room
function resetRoom() {
    clearInterval(interval);
    initRoom();
    drawRoom();
}

// Init
initRoom();
drawRoom();
</script>
